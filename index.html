🎯 원하는 기능을 선택하세요

🗑️ 내 트윗 삭제

내가 작성한 모든 트윗을 안전하게 삭제합니다

javascript:(function(){let count=0;const tweets=document.querySelectorAll('[data-testid="tweet"]');if(tweets.length===0){alert('트윗을 찾을 수 없습니다. 트위터 프로필 페이지로 이동해주세요.');return;}const myTweets=Array.from(tweets).filter(tweet=>{const retweetIcon=tweet.querySelector('[data-testid="retweet"]');const quoteIcon=tweet.querySelector('[data-testid="quote"]');return !retweetIcon?.getAttribute('aria-label')?.includes('리트윗 취소') && !quoteIcon;});if(myTweets.length===0){alert('삭제할 트윗을 찾을 수 없습니다.');return;}if(!confirm('내가 쓴 트윗 '+myTweets.length+'개를 삭제하시겠습니까?\\n\\n⚠️ 이 작업은 되돌릴 수 없습니다!')){return;}let processed=0;myTweets.forEach((tweet,i)=>{setTimeout(()=>{try{const moreBtn=tweet.querySelector('[aria-label="More"]')||tweet.querySelector('[aria-label="더보기"]');if(moreBtn){moreBtn.click();setTimeout(()=>{const deleteBtn=Array.from(document.querySelectorAll('[role="menuitem"]')).find(btn=>btn.textContent.includes('Delete')||btn.textContent.includes('삭제'));if(deleteBtn){deleteBtn.click();setTimeout(()=>{const confirmBtn=document.querySelector('[data-testid="confirmationSheetConfirm"]');if(confirmBtn){confirmBtn.click();processed++;console.log('✅ 트윗 삭제: '+processed+'/'+myTweets.length);if(processed===myTweets.length){alert('🎉 내 트윗 삭제 완료!\\n삭제된 트윗: '+processed+'개');}}},1000);}},1000);}}catch(e){console.error('❌ 에러:',e);}},i*2500);});alert('🚀 내 트윗 삭제 시작!\\n총 '+myTweets.length+'개 처리 예정');})();

❤️ 좋아요 취소

내가 누른 모든 좋아요를 한 번에 취소합니다

javascript:(function(){let count=0;const likeButtons=document.querySelectorAll('[data-testid="like"]');const activeLikes=Array.from(likeButtons).filter(btn=>btn.getAttribute('aria-label')?.includes('좋아요 취소')||btn.querySelector('svg')?.style?.color==='rgb(249, 24, 128)');if(activeLikes.length===0){alert('취소할 좋아요를 찾을 수 없습니다.\\n좋아요 목록 페이지에서 실행해주세요.');return;}if(!confirm('좋아요 '+activeLikes.length+'개를 모두 취소하시겠습니까?\\n\\n⚠️ 이 작업은 되돌릴 수 없습니다!')){return;}let processed=0;activeLikes.forEach((likeBtn,i)=>{setTimeout(()=>{try{if(likeBtn.getAttribute('aria-label')?.includes('좋아요 취소')||likeBtn.querySelector('svg')?.style?.color==='rgb(249, 24, 128)'){likeBtn.click();processed++;console.log('💔 좋아요 취소: '+processed+'/'+activeLikes.length);if(processed===activeLikes.length){alert('🎉 좋아요 취소 완료!\\n취소된 좋아요: '+processed+'개');}}}catch(e){console.error('❌ 에러:',e);}},i*1500);});alert('💔 좋아요 취소 시작!\\n총 '+activeLikes.length+'개 처리 예정');})();

🔄 리트윗 취소

내가 리트윗한 모든 게시물을 취소합니다

javascript:(function(){let count=0;const retweetButtons=document.querySelectorAll('[data-testid="retweet"]');const activeRetweets=Array.from(retweetButtons).filter(btn=>btn.getAttribute('aria-label')?.includes('리트윗 취소')||btn.querySelector('svg')?.style?.color==='rgb(0, 186, 124)');if(activeRetweets.length===0){alert('취소할 리트윗을 찾을 수 없습니다.\\n프로필 페이지에서 실행해주세요.');return;}if(!confirm('리트윗 '+activeRetweets.length+'개를 모두 취소하시겠습니까?\\n\\n⚠️ 이 작업은 되돌릴 수 없습니다!')){return;}let processed=0;activeRetweets.forEach((retweetBtn,i)=>{setTimeout(()=>{try{if(retweetBtn.getAttribute('aria-label')?.includes('리트윗 취소')||retweetBtn.querySelector('svg')?.style?.color==='rgb(0, 186, 124)'){retweetBtn.click();setTimeout(()=>{const confirmBtn=document.querySelector('[data-testid="unretweetConfirm"]');if(confirmBtn){confirmBtn.click();processed++;console.log('🔄 리트윗 취소: '+processed+'/'+activeRetweets.length);if(processed===activeRetweets.length){alert('🎉 리트윗 취소 완료!\\n취소된 리트윗: '+processed+'개');}}},800);}}catch(e){console.error('❌ 에러:',e);}},i*2000);});alert('🔄 리트윗 취소 시작!\\n총 '+activeRetweets.length+'개 처리 예정');})();

💬 인용트윗 삭제

javascript:(function(){let count=0;const tweets=document.querySelectorAll('[data-testid="tweet"]');const quoteTweets=Array.from(tweets).filter(tweet=>{const quotedTweet=tweet.querySelector('[role="link"]');const hasQuoteStructure=tweet.textContent.length>0&"edTweet&&tweet.querySelector('[data-testid="User-Name"]');return hasQuoteStructure;});if(quoteTweets.length===0){alert('삭제할 인용트윗을 찾을 수 없습니다.\\n프로필 페이지에서 실행해주세요.');return;}if(!confirm('인용트윗 '+quoteTweets.length+'개를 삭제하시겠습니까?\\n\\n⚠️ 이 작업은 되돌릴 수 없습니다!')){return;}let processed=0;quoteTweets.forEach((tweet,i)=>{setTimeout(()=>{try{const moreBtn=tweet.querySelector('[aria-label="More"]')||tweet.querySelector('[aria-label="더보기"]');if(moreBtn){moreBtn.click();setTimeout(()=>{const deleteBtn=Array.from(document.querySelectorAll('[role="menuitem"]')).find(btn=>btn.textContent.includes('Delete')||btn.textContent.includes('삭제'));if(deleteBtn){deleteBtn.click();setTimeout(()=>{const confirmBtn=document.querySelector('[data-testid="confirmationSheetConfirm"]');if(confirmBtn){confirmBtn.click();processed++;console.log('💬 인용트윗 삭제: '+processed+'/'+quoteTweets.length);if(processed===quoteTweets.length){alert('🎉 인용트윗 삭제 완료!\\n삭제된 인용트윗: '+processed+'개');}}},1000);}},1000);}}catch(e){console.error('❌ 에러:',e);}},i*2500);});alert('💬 인용트윗 삭제 시작!\\n총 '+quoteTweets.length+'개 처리 예정');})();

📖 사용 방법 가이드

1️⃣ 코드 복사

  • 원하는 기능의 코드 복사 버튼 클릭
  • 자동으로 클립보드에 복사됨

2️⃣ 북마크 생성

  • 브라우저 북마크바 우클릭
  • "새 북마크 추가" 선택
  • 이름: 기능명 입력
  • URL: 복사한 코드 붙여넣기

3️⃣ 해당 페이지 이동

  • 트윗 삭제: 내 프로필
  • 좋아요 취소: 좋아요 목록
  • 리트윗 취소: 내 프로필
  • 인용트윗 삭제: 내 프로필

4️⃣ 북마클릿 실행

  • 생성한 북마크 클릭
  • 확인 창에서 "확인" 선택
  • 자동 처리 진행 확인

💡 전문가 팁

📍 올바른 페이지에서 실행: 각 기능은 해당하는 트위터 페이지에서만 작동합니다. 좋아요 취소는 좋아요 목록에서, 트윗 삭제는 프로필에서 실행하세요.
🔍 진행상황 확인: F12 키를 눌러 개발자 도구를 열고 Console 탭에서 실시간으로 진행상황을 확인할 수 있습니다.
⏸️ 중도 중단: 작업을 중단하려면 페이지를 새로고침(F5)하거나 다른 페이지로 이동하세요.
🌐 WiFi 권장: 대량의 데이터를 처리하므로 안정적인 WiFi 환경에서 사용하시기 바랍니다.

⚠️ 중요한 주의사항

  • 되돌릴 수 없음: 모든 삭제/취소 작업은 영구적이며 복구할 수 없습니다
  • 계정 안전: 너무 빠른 대량 작업은 일시적인 계정 제재를 받을 수 있습니다
  • 백업 권장: 중요한 데이터는 미리 백업해 두시기 바랍니다
  • 책임 소재: 본 도구 사용으로 발생하는 모든 결과는 사용자 책임입니다
  • 트위터 정책: 트위터 이용약관을 준수하며 사용하세요
