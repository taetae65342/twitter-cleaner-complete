🧹 트위터 완전 청소기

트위터 웹 페이지에서 개발자 도구(F12 → Console)에 붙여넣기만 하면 작동

💡 사용 방법

  1. 원하는 기능의 복사 버튼을 클릭해서 코드를 복사하세요
  2. 트위터 웹사이트(twitter.com)로 이동하세요
  3. 각 기능에 표시된 실행 페이지로 이동하세요
  4. F12를 눌러 개발자 도구를 열고 Console 탭으로 이동하세요
  5. 복사한 코드를 Console에 붙여넣고 Enter를 누르세요
  6. 스크립트가 자동으로 실행됩니다

좋아요 취소

좋아요한 모든 트윗을 최신부터 순서대로 취소합니다

📍 실행 페이지: twitter.com/i/likes

JavaScript Code
(async function() { console.log("❤️ 좋아요 취소 시작"); if (!confirm("좋아요를 취소하시겠습니까?")) return; window.scrollTo({top: 0, behavior: "smooth"}); await new Promise(r => setTimeout(r, 2000)); let count = 0; for (let i = 0; i < 50; i++) { const buttons = document.querySelectorAll('[data-testid="unlike"]'); if (buttons.length > 0) { for (const btn of buttons) { if (btn && btn.offsetParent) { btn.click(); count++; console.log(`💔 완료: ${count}개`); await new Promise(r => setTimeout(r, 500)); } } } window.scrollBy(0, window.innerHeight * 0.8); await new Promise(r => setTimeout(r, 2000)); if (window.innerHeight + window.scrollY >= document.body.scrollHeight - 100) break; } alert(`🎉 완료! ${count}개 취소됨`); })();

내 트윗 삭제

모든 내 트윗을 최신부터 순서대로 삭제합니다

📍 실행 페이지: twitter.com/사용자명

JavaScript Code
(async function() { console.log("🗑️ 내 트윗 삭제 시작"); if (!confirm("내 트윗을 삭제하시겠습니까?")) return; window.scrollTo({top: 0, behavior: "smooth"}); await new Promise(r => setTimeout(r, 2000)); let count = 0; for (let i = 0; i < 50; i++) { const tweets = Array.from(document.querySelectorAll('[data-testid="tweet"]')).filter(t => { const isRetweet = t.querySelector('[data-testid="socialContext"]'); const isQuote = t.querySelector('[data-testid="quoteTweet"]'); return !isRetweet && !isQuote && t.offsetParent; }); for (const tweet of tweets) { try { const moreBtn = tweet.querySelector('[aria-label="더보기"]'); if (moreBtn) { moreBtn.click(); await new Promise(r => setTimeout(r, 800)); const deleteBtn = [...document.querySelectorAll('[role="menuitem"]')] .find(b => b.textContent.includes("삭제")); if (deleteBtn) { deleteBtn.click(); await new Promise(r => setTimeout(r, 800)); const confirmBtn = document.querySelector('[data-testid="confirmationSheetConfirm"]'); if (confirmBtn) { confirmBtn.click(); count++; console.log(`🗑️ 완료: ${count}개`); await new Promise(r => setTimeout(r, 1000)); } } } } catch (e) { console.error("오류:", e); } } window.scrollBy(0, window.innerHeight * 0.8); await new Promise(r => setTimeout(r, 3500)); if (window.innerHeight + window.scrollY >= document.body.scrollHeight - 100) break; } alert(`🎉 완료! ${count}개 삭제됨`); })();
