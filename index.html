🧹 트위터 완전 청소기

트위터 웹 페이지에서 개발자 도구(F12 → Console)에 붙여넣기만 하면 작동

💡 사용 방법

  1. 원하는 기능의 복사 버튼을 클릭해서 코드를 복사하세요
  2. 트위터 웹사이트(twitter.com)로 이동하세요
  3. 각 기능에 표시된 실행 페이지로 이동하세요
  4. F12를 눌러 개발자 도구를 열고 Console 탭으로 이동하세요
  5. 복사한 코드를 Console에 붙여넣고 Enter를 누르세요
  6. 스크립트가 자동으로 실행됩니다

좋아요 취소

자동 스크롤로 모든 좋아요를 불러온 후 일괄 취소합니다

📍 실행 페이지: twitter.com/i/likes

JavaScript Code
(async function cancelLikes() { console.log("❤️ 좋아요 취소 + 동시 스크롤 시작!"); if (!confirm("좋아요를 스크롤하면서 동시에 취소하시겠습니까?\n\n⚠️ 이 작업은 되돌릴 수 없습니다!")) { return; } let canceledCount = 0; let scrollCount = 0; const maxScrolls = 50; const scrollInterval = 2000; for (let i = 0; i < maxScrolls; i++) { // 스크롤 실행 window.scrollTo({top: document.body.scrollHeight, behavior: "smooth"}); scrollCount++; console.log(`📜 스크롤 진행: ${scrollCount}/${maxScrolls}`); // 스크롤 후 잠시 대기 (DOM 로딩 시간) await new Promise(resolve => setTimeout(resolve, 1000)); // 현재 화면에 있는 좋아요 버튼들 찾기 const likeButtons = document.querySelectorAll('[data-testid="unlike"]'); if (likeButtons.length > 0) { console.log(`💔 현재 화면에서 ${likeButtons.length}개 좋아요 발견!`); // 발견된 좋아요들 즉시 취소 for (let j = 0; j < likeButtons.length; j++) { try { const button = likeButtons[j]; if (button && button.offsetParent !== null) { // 화면에 보이는 버튼만 button.click(); canceledCount++; console.log(`💔 좋아요 취소 완료: ${canceledCount}개`); await new Promise(resolve => setTimeout(resolve, 300)); } } catch (error) { console.error("❌ 취소 중 오류:", error); } } } // 다음 스크롤까지 대기 await new Promise(resolve => setTimeout(resolve, scrollInterval - 1000)); // 더 이상 스크롤할 내용이 없으면 중단 if (window.innerHeight + window.scrollY >= document.body.scrollHeight - 100) { console.log("📜 페이지 끝에 도달했습니다."); break; } } // 마지막으로 남은 좋아요들 처리 console.log("🔍 마지막 좋아요 확인 중..."); const remainingLikes = document.querySelectorAll('[data-testid="unlike"]'); if (remainingLikes.length > 0) { console.log(`💔 마지막 ${remainingLikes.length}개 좋아요 취소 중...`); for (let i = 0; i < remainingLikes.length; i++) { try { const button = remainingLikes[i]; if (button) { button.click(); canceledCount++; console.log(`💔 좋아요 취소: ${canceledCount}개`); await new Promise(resolve => setTimeout(resolve, 500)); } } catch (error) { console.error("❌ 취소 중 오류:", error); } } } alert(`🎉 좋아요 취소 완료!\n총 취소된 좋아요: ${canceledCount}개\n스크롤 횟수: ${scrollCount}회`); })();

인용트윗 삭제

자동 스크롤로 모든 인용트윗을 불러온 후 삭제합니다

📍 실행 페이지: twitter.com/사용자명

JavaScript Code
(async function deleteQuoteTweets() { console.log("💬 인용트윗 삭제 + 동시 스크롤 시작!"); if (!confirm("인용트윗을 스크롤하면서 동시에 삭제하시겠습니까?\n\n⚠️ 이 작업은 되돌릴 수 없습니다!")) { return; } let deletedCount = 0; let scrollCount = 0; const maxScrolls = 50; const scrollInterval = 3000; for (let i = 0; i < maxScrolls; i++) { // 스크롤 실행 window.scrollTo({top: document.body.scrollHeight, behavior: "smooth"}); scrollCount++; console.log(`📜 스크롤 진행: ${scrollCount}/${maxScrolls}`); // 스크롤 후 잠시 대기 (DOM 로딩 시간) await new Promise(resolve => setTimeout(resolve, 1500)); // 현재 화면에 있는 인용트윗들 찾기 const allTweets = document.querySelectorAll('[data-testid="tweet"]'); const quoteTweets = Array.from(allTweets).filter(tweet => { return tweet.querySelector('[data-testid="quoteTweet"]') || (tweet.innerText.includes("https://t.co") && tweet.offsetParent !== null); }); if (quoteTweets.length > 0) { console.log(`💬 현재 화면에서 ${quoteTweets.length}개 인용트윗 발견!`); // 발견된 인용트윗들 즉시 삭제 for (let j = 0; j < quoteTweets.length; j++) { try { const tweet = quoteTweets[j]; if (tweet && tweet.offsetParent !== null) { // 화면에 보이는 트윗만 const moreBtn = tweet.querySelector('[aria-label="더보기"]') || tweet.querySelector('[aria-label="More"]'); if (moreBtn) { moreBtn.click(); await new Promise(resolve => setTimeout(resolve, 800)); const deleteBtn = [...document.querySelectorAll('[role="menuitem"]')] .find(btn => btn.textContent.includes("삭제") || btn.textContent.includes("Delete")); if (deleteBtn) { deleteBtn.click(); await new Promise(resolve => setTimeout(resolve, 800)); const confirmBtn = document.querySelector('[data-testid="confirmationSheetConfirm"]'); if (confirmBtn) { confirmBtn.click(); deletedCount++; console.log(`💬 인용트윗 삭제 완료: ${deletedCount}개`); await new Promise(resolve => setTimeout(resolve, 500)); } } } } } catch (error) { console.error("❌ 삭제 중 오류:", error); } } } // 다음 스크롤까지 대기 await new Promise(resolve => setTimeout(resolve, scrollInterval - 1500)); // 더 이상 스크롤할 내용이 없으면 중단 if (window.innerHeight + window.scrollY >= document.body.scrollHeight - 100) { console.log("📜 페이지 끝에 도달했습니다."); break; } } alert(`🎉 인용트윗 삭제 완료!\n총 삭제된 인용트윗: ${deletedCount}개\n스크롤 횟수: ${scrollCount}회`); })();
