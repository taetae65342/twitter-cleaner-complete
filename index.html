🧹 트위터 완전 청소기

트위터 웹 페이지에서 개발자 도구(F12 → Console)에 붙여넣기만 하면 작동

💡 사용 방법

  1. 원하는 기능의 복사 버튼을 클릭해서 코드를 복사하세요
  2. 트위터 웹사이트(twitter.com)로 이동하세요
  3. 각 기능에 표시된 실행 페이지로 이동하세요
  4. F12를 눌러 개발자 도구를 열고 Console 탭으로 이동하세요
  5. 복사한 코드를 Console에 붙여넣고 Enter를 누르세요
  6. 스크립트가 자동으로 실행됩니다

권장: 크롬 브라우저 + PC만 가능

좋아요 취소

자동 스크롤로 모든 좋아요를 불러온 후 일괄 취소합니다

📍 실행 페이지: twitter.com/i/likes

JavaScript Code
(async function cancelLikes() { console.log("❤️ 좋아요 취소 + 최신부터 시작!"); if (!confirm("좋아요를 최신부터 순서대로 취소하시겠습니까?\n\n⚠️ 이 작업은 되돌릴 수 없습니다!")) { return; } // 페이지 맨 위로 이동 window.scrollTo({top: 0, behavior: "smooth"}); await new Promise(resolve => setTimeout(resolve, 2000)); let canceledCount = 0; let scrollCount = 0; const maxScrolls = 50; let previousHeight = 0; for (let i = 0; i < maxScrolls; i++) { scrollCount++; console.log(`📜 진행 상황: ${scrollCount}/${maxScrolls}`); // 현재 화면에 있는 좋아요 버튼들 찾기 const likeButtons = document.querySelectorAll('[data-testid="unlike"]'); if (likeButtons.length > 0) { console.log(`💔 현재 화면에서 ${likeButtons.length}개 좋아요 발견!`); // 발견된 좋아요들 즉시 취소 for (let j = 0; j < likeButtons.length; j++) { try { const button = likeButtons[j]; if (button && button.offsetParent !== null) { // 화면에 보이는 버튼만 button.click(); canceledCount++; console.log(`💔 좋아요 취소 완료: ${canceledCount}개 (최신부터)`); await new Promise(resolve => setTimeout(resolve, 300)); } } catch (error) { console.error("❌ 취소 중 오류:", error); } } } // 조금씩 아래로 스크롤 (최신에서 오래된 순서) const currentHeight = window.scrollY; window.scrollTo({top: currentHeight + window.innerHeight * 0.8, behavior: "smooth"}); await new Promise(resolve => setTimeout(resolve, 2000)); // 더 이상 스크롤할 내용이 없으면 중단 if (window.innerHeight + window.scrollY >= document.body.scrollHeight - 100) { console.log("📜 페이지 끝에 도달했습니다."); break; } // 스크롤이 더 이상 진행되지 않으면 중단 if (Math.abs(window.scrollY - previousHeight) < 50) { console.log("📜 더 이상 스크롤할 수 없습니다."); break; } previousHeight = window.scrollY; } alert(`🎉 좋아요 취소 완료! (최신부터 순서대로)\n총 취소된 좋아요: ${canceledCount}개\n스크롤 횟수: ${scrollCount}회`); })();
