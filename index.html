🧹 트위터 완전 청소기

⌨️ Console 실행 버전

트위터에서 F12 → Console에 코드를 붙여넣기만 하면 바로 실행!

⚠️ 주의사항

아직 개발단계이므로 실사용은 아직 불가합니다

사용으로 인한 손해는 책임지지 않습니다

🚧 테스트 목적으로만 사용하시고, 중요한 데이터 백업을 권장합니다

🚀 사용 방법 (3단계)

1️⃣ 트위터 페이지 이동

해당하는 트위터 페이지로 이동하세요

2️⃣ 개발자 도구 열기

F12 키를 누르고 Console 탭 클릭

3️⃣ 코드 붙여넣기

아래 코드를 복사해서 Console에 붙여넣기 후 Enter

🎯 원하는 기능을 선택하세요

🗑️ 내 트윗 삭제 + 무한 스크롤

무한 스크롤로 모든 트윗을 연속적으로 불러와서 안전하게 삭제합니다

📍 실행 페이지: 내 프로필 (twitter.com/사용자명)
(() => { alert("🗑️ 무한 트윗 삭제 시작!"); let c = 0; let running = true; function scrollAndDelete() { window.scrollTo({ top: document.body.scrollHeight, behavior: "smooth" }); setTimeout(() => { const tweets = [...document.querySelectorAll('[data-testid="tweet"]')]; console.log(`현재 라운드에서 찾은 트윗: ${tweets.length}개`); tweets.forEach((t, i) => { setTimeout(() => { try { let more = t.querySelector('[aria-label="더보기"]') || t.querySelector('[aria-label="More"]'); more?.click(); setTimeout(() => { let del = [...document.querySelectorAll('[role="menuitem"]')].find(b => b.textContent.includes("삭제") || b.textContent.includes("Delete")); del?.click(); setTimeout(() => { let confirm = document.querySelector('[data-testid="confirmationSheetConfirm"]'); confirm?.click(); c++; console.log(`✅ 삭제됨: ${c}개`); }, 500); }, 500); } catch (e) { console.log("❌ 삭제 중 오류:", e); } }, i * 1500); }); if (running) { setTimeout(scrollAndDelete, 8000); // 다음 라운드 } }, 1500); } // 중단 방법 안내 console.log("🛑 중단하려면 다음 명령어를 실행하세요: running = false"); scrollAndDelete(); })();

❤️ 좋아요 취소

자동 스크롤로 모든 좋아요를 불러온 후 일괄 취소합니다

📍 실행 페이지: 좋아요 목록 (twitter.com/i/likes)
(async function cancelLikes() { console.log("❤️ 좋아요 취소 + 자동 스크롤 시작!"); let scrollCount = 0; const maxScrolls = 50; const scrollInterval = 1500; // 자동 스크롤로 좋아요 로드 console.log("📜 자동 스크롤 시작..."); for (let i = 0; i < maxScrolls; i++) { window.scrollTo({top: document.body.scrollHeight, behavior: "smooth"}); console.log(`스크롤 진행: ${i + 1}/${maxScrolls}`); await new Promise(resolve => setTimeout(resolve, scrollInterval)); } // 좋아요 취소 시작 console.log("💔 좋아요 취소 시작!"); const likeButtons = document.querySelectorAll('[data-testid="unlike"]'); console.log(`찾은 좋아요 개수: ${likeButtons.length}개`); if (likeButtons.length === 0) { alert("❌ 취소할 좋아요를 찾을 수 없습니다.\n좋아요 목록 페이지에서 실행해주세요."); return; } if (!confirm(`${likeButtons.length}개의 좋아요를 취소하시겠습니까?\n\n⚠️ 이 작업은 되돌릴 수 없습니다!`)) { return; } let canceledCount = 0; for (let i = 0; i < likeButtons.length; i++) { try { const button = likeButtons[i]; if (button) { button.click(); canceledCount++; console.log(`💔 좋아요 취소: ${canceledCount}/${likeButtons.length}`); await new Promise(resolve => setTimeout(resolve, 1500)); } } catch (error) { console.error("❌ 취소 중 오류:", error); } } alert(`🎉 좋아요 취소 완료!\n취소된 좋아요: ${canceledCount}개`); })();

🔄 리트윗 취소

자동 스크롤로 모든 리트윗을 불러온 후 일괄 취소합니다

📍 실행 페이지: 내 프로필 (twitter.com/사용자명)
(async function cancelRetweets() { console.log("🔄 리트윗 취소 + 자동 스크롤 시작!"); let scrollCount = 0; const maxScrolls = 50; const scrollInterval = 1500; // 자동 스크롤로 리트윗 로드 console.log("📜 자동 스크롤 시작..."); for (let i = 0; i < maxScrolls; i++) { window.scrollTo({top: document.body.scrollHeight, behavior: "smooth"}); console.log(`스크롤 진행: ${i + 1}/${maxScrolls}`); await new Promise(resolve => setTimeout(resolve, scrollInterval)); } // 리트윗 취소 시작 console.log("🔄 리트윗 취소 시작!"); const retweetButtons = document.querySelectorAll('[data-testid="unretweet"]'); console.log(`찾은 리트윗 개수: ${retweetButtons.length}개`); if (retweetButtons.length === 0) { alert("❌ 취소할 리트윗을 찾을 수 없습니다.\n프로필 페이지에서 실행해주세요."); return; } if (!confirm(`${retweetButtons.length}개의 리트윗을 취소하시겠습니까?\n\n⚠️ 이 작업은 되돌릴 수 없습니다!`)) { return; } let canceledCount = 0; for (let i = 0; i < retweetButtons.length; i++) { try { const button = retweetButtons[i]; if (button) { button.click(); await new Promise(resolve => setTimeout(resolve, 800)); const confirmBtn = document.querySelector('[data-testid="unretweetConfirm"]'); if (confirmBtn) { confirmBtn.click(); canceledCount++; console.log(`🔄 리트윗 취소: ${canceledCount}/${retweetButtons.length}`); await new Promise(resolve => setTimeout(resolve, 2000)); } } } catch (error) { console.error("❌ 취소 중 오류:", error); } } alert(`🎉 리트윗 취소 완료!\n취소된 리트윗: ${canceledCount}개`); })();

💬 인용트윗 삭제

자동 스크롤로 모든 인용트윗을 불러온 후 삭제합니다

📍 실행 페이지: 내 프로필 (twitter.com/사용자명)
(async function deleteQuoteTweets() { console.log("💬 인용트윗 삭제 + 자동 스크롤 시작!"); let scrollCount = 0; const maxScrolls = 50; const scrollInterval = 1500; // 자동 스크롤로 트윗 로드 console.log("📜 자동 스크롤 시작..."); for (let i = 0; i < maxScrolls; i++) { window.scrollTo({top: document.body.scrollHeight, behavior: "smooth"}); console.log(`스크롤 진행: ${i + 1}/${maxScrolls}`); await new Promise(resolve => setTimeout(resolve, scrollInterval)); } // 인용트윗 찾기 및 삭제 console.log("💬 인용트윗 삭제 시작!"); const allTweets = document.querySelectorAll('[data-testid="tweet"]'); const quoteTweets = Array.from(allTweets).filter(tweet => { return tweet.innerText.includes("https://t.co") || tweet.querySelector('[data-testid="quoteTweet"]'); }); console.log(`찾은 인용트윗 개수: ${quoteTweets.length}개`); if (quoteTweets.length === 0) { alert("❌ 삭제할 인용트윗을 찾을 수 없습니다."); return; } if (!confirm(`${quoteTweets.length}개의 인용트윗을 삭제하시겠습니까?\n\n⚠️ 이 작업은 되돌릴 수 없습니다!`)) { return; } let deletedCount = 0; for (let i = 0; i < quoteTweets.length; i++) { try { const tweet = quoteTweets[i]; const moreBtn = tweet.querySelector('[aria-label="더보기"]') || tweet.querySelector('[aria-label="More"]'); if (moreBtn) { moreBtn.click(); await new Promise(resolve => setTimeout(resolve, 1000)); const deleteBtn = [...document.querySelectorAll('[role="menuitem"]')] .find(btn => btn.textContent.includes("삭제") || btn.textContent.includes("Delete")); if (deleteBtn) { deleteBtn.click(); await new Promise(resolve => setTimeout(resolve, 1000)); const confirmBtn = document.querySelector('[data-testid="confirmationSheetConfirm"]'); if (confirmBtn) { confirmBtn.click(); deletedCount++; console.log(`💬 인용트윗 삭제: ${deletedCount}/${quoteTweets.length}`); await new Promise(resolve => setTimeout(resolve, 2500)); } } } } catch (error) { console.error("❌ 삭제 중 오류:", error); } } alert(`🎉 인용트윗 삭제 완료!\n삭제된 인용트윗: ${deletedCount}개`); })();

💡 Console 사용 팁

  • 진행상황 확인: Console에서 실시간으로 진행상황을 볼 수 있습니다
  • 무한 삭제 중단: Console에서 running = false 입력 후 Enter
  • 페이지 새로고침: F5키로 언제든 모든 작업을 중단할 수 있습니다
  • 안전한 속도: 자동으로 안전한 간격을 두고 처리합니다
  • 오류 처리: 오류 발생시 자동으로 건너뛰고 계속 진행합니다
  • 완료 알림: 각 라운드별 처리 현황을 실시간으로 알려줍니다

⚠️ 중요한 주의사항

  • 되돌릴 수 없음: 모든 삭제/취소 작업은 영구적이며 복구할 수 없습니다
  • 테스트 권장: 중요한 계정에서는 소량으로 먼저 테스트해보세요
  • 백업 필수: 중요한 데이터는 미리 백업해 두시기 바랍니다
  • 계정 안전: 너무 빠른 대량 작업은 일시적인 계정 제재를 받을 수 있습니다
  • 개발 단계: 아직 개발 중인 도구이므로 신중하게 사용하세요
